1function output_signal = simple_compressor(input_signal)
 2
 3global gaintable;
 4global attack_coeffs;
 5global release_coeffs;
 6global gaintable_levels;
 7
 8persistent state;
 9
10nchannels_in = size(input_signal,2);
11%fragsize = size(input_signal,1);
12
13if isempty(state)
14    state = ones(1,nchannels_in)*65;
15end
16
17assert(size(gaintable,1) == nchannels_in);
18assert(size(attack_coeffs,2) == nchannels_in);
19assert(size(attack_coeffs,1) == 1);
20assert(isequal(size(release_coeffs), size(attack_coeffs)));
21assert(size(gaintable_levels,1) == 1);
22assert(size(gaintable_levels,2) == size(gaintable,2));
23
24meansquares = mean(input_signal.^2);
25instant_levels = 10*log10(meansquares / 4e-10 + 1e-10);
26filtered_levels = state;
27
28output_signal = input_signal;
29for channel = 1:nchannels_in
30    coeff = release_coeffs(channel);
31    if (instant_levels(channel) > filtered_levels(channel))
32        coeff = attack_coeffs(channel);
33    end
34    filtered_levels(channel) = filtered_levels(channel) * coeff + (1-coeff) * instant_levels(channel);
35    gain = interp1(gaintable_levels, gaintable(channel,:), filtered_levels(channel));
36    output_signal(:,channel) = input_signal(:,channel) * 10.^(gain/20);
37end
38
39state = filtered_levels;
